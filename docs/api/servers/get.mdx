## Endpoint

<Card>
  <code>GET https://api.superbox.ai/api/v1/servers/:name</code>
</Card>

## Authentication

<Tabs>
  <Tab title="Optional">
    This endpoint works with or without authentication.
    <Info>
      Authenticated requests have access to additional details and higher rate
      limits
    </Info>
  </Tab>
</Tabs>

## Path Parameters

<ParamField path="name" type="string" required>
  Unique server identifier (lowercase with hyphens) **Example:** `weather-mcp`,
  `database-query-mcp`
</ParamField>

```python
Python
import requests

# Get server details
server_name = "weather-mcp"
url = f"https://api.superbox.ai/api/v1/servers/{server_name}"

response = requests.get(url)
server = response.json()['data']

print(f"{server['name']} v{server['version']}")
print(f"Author: {server['author']}")
print(f"Language: {server['lang']}")
print(f"Tools: {', '.join(server['tools']['names'])}")

# With authentication
headers = {
    'Authorization': f'Bearer {os.getenv("SUPERBOX_API_TOKEN")}',
    'Content-Type': 'application/json'
}

response = requests.get(url, headers=headers)
server = response.json()['data']

# Check pricing
if server['pricing']['amount'] == 0:
    print("This server is FREE!")
else:
    print(f"Price: {server['pricing']['currency']} {server['pricing']['amount']}")
```

```go
Go
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
    "os"
)

func main() {
    serverName := "weather-mcp"
    url := fmt.Sprintf("https://api.superbox.ai/api/v1/servers/%s", serverName)

    // Create request
    req, err := http.NewRequest("GET", url, nil)
    if err != nil {
        panic(err)
    }

    // Add authentication (optional)
    token := os.Getenv("SUPERBOX_API_TOKEN")
    if token != "" {
        req.Header.Set("Authorization", "Bearer "+token)
    }
    req.Header.Set("Content-Type", "application/json")

    // Make request
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    // Parse response
    var result struct {
        Success bool `json:"success"`
        Data    struct {
            Name    string `json:"name"`
            Version string `json:"version"`
            Author  string `json:"author"`
        } `json:"data"`
    }

    json.NewDecoder(resp.Body).Decode(&result)

    fmt.Printf("%s v%s by %s\n",
        result.Data.Name,
        result.Data.Version,
        result.Data.Author)
}
```

    def is_server_secure(server_name):
        response = requests.get(
            f'https://api.superbox.ai/api/v1/servers/{server_name}'
        )
        server = response.json()['data']
        
        security = server.get('security_report', {}).get('summary', {})

## Understanding Security Reports

The `security_report` object provides comprehensive security analysis:

<Tabs>
  <Tab title="SonarQube">
    **Code Quality Analysis** - **Bugs**: Actual coding errors -
    **Vulnerabilities**: Security vulnerabilities - **Code Smells**:
    Maintainability issues - **Coverage**: Test coverage percentage - **Quality
    Gate**: Overall pass/fail status **Ratings (A-E):** - A: Excellent - B: Good
    - C: Acceptable - D: Poor - E: Critical issues
  </Tab>

<Tab title="GitGuardian">
  **Secrets Detection** Scans for exposed credentials: - API keys - Access
  tokens - Passwords - Private keys - Database credentials `scan_passed: true`
  means no secrets found
</Tab>

  <Tab title="Bandit">
    **Python Security Audit** (Python servers only) - **High Severity**:
    Critical security flaws - **Medium Severity**: Potential vulnerabilities -
    **Low Severity**: Minor security concerns Checks for: - Unsafe function
    usage - SQL injection risks - Command injection risks - Cryptography issues
  </Tab>
</Tabs>
